注册

1.SendBroadcastActivity.java

public class SendBroadcastActivity extends Activity {

public static final String COM_GENTLENI_TEST_BROADCAST = "com.gentleni.test.broadcast";
private TestBroadcastReceiver mTestBroadcastReceiver = new TestBroadcastReceiver();
private Button mButtonSendBroadcast;
@Override
protected void onCreate(Bundle savedInstanceState) {

    setContentView(R.layout.activity_sendbroadcast);
    mButtonSendBroadcast=(Button)findViewById(R.id.btn_send_broadcast);

    mButtonSendBroadcast.setOnClickListener(new View.OnClickListener() {
        @Override
        public void onClick(View v) {

            // 发送广播
            Intent intent = new Intent();
            intent.setAction(COM_GENTLENI_TEST_BROADCAST);
            intent.putExtra("toast", "this is my toast of broadcast");

            sendBroadcast(intent);
        }
    });
    super.onCreate(savedInstanceState);
}

@Override
protected void onStart() {
    super.onStart();
//        动态注册BroadcastReceiver
    IntentFilter intentFilter = new IntentFilter();
    intentFilter.addAction(COM_GENTLENI_TEST_BROADCAST);
    registerReceiver(mTestBroadcastReceiver, intentFilter);
}
}

2.BroadcastReceiver.java

public class TestBroadcastReceiver extends BroadcastReceiver{
    @Override
    public void onReceive(Context context, Intent intent) {

        if (intent != null) {
            if (TextUtils.equals(intent.getAction(), SendBroadcastActivity.COM_GENTLENI_TEST_BROADCAST)) {
                String toastString = intent.getStringExtra("toast");
                Toast.makeText(context, toastString, Toast.LENGTH_LONG).show();
            }
        }
    }
}

3.AndroidManifest.xml

<receiver android:name=".TestBroadcastReceiver">
           <intent-filter>
               <action android:name="com.gentleni.test.broadcast"></action>
           </intent-filter>
       </receiver>
